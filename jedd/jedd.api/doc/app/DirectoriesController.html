def index
        ##
        # Send root dir
        root_directory = User.find(permitted_params[:profile_id]).jedd_directories.where(:jedd_directory_id => nil).first
        @directory = []
        @directory.concat(root_directory.jedd_directories)
        @directory.concat(root_directory.jedd_file_nodes)
end
